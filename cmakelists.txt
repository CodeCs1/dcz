cmake_minimum_required(VERSION 3.10)
project(dcz)
enable_language(CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)


include(FetchContent)

FetchContent_Declare(
    CLI11
    GIT_REPOSITORY https://github.com/CLIUtils/CLI11.git
    GIT_TAG main
)
FetchContent_MakeAvailable(CLI11)

target_include_directories (CLI11 INTERFACE "${CMAKE_CURRENT_LIST_DIR}/include")


file(GLOB AST "AST/*.cpp")
file(GLOB ERROR_HANDLER "ErrorHandler/*.cpp")

file(GLOB AST2MC "AST2MC/*.cpp" "AST2MC/codegen/*.cpp")


add_executable(dcz ${AST}  ${ERROR_HANDLER} ${AST2MC} main.cpp)

target_include_directories(dcz PUBLIC "include")
